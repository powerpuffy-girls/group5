FILENAME = "my_diary.txt"

def create_file():
    try:
        file = open(FILENAME, "x") 
        file.close()
        print("Diary file created!\n")
    except FileExistsError:
        print("Diary file already exists.\n")

def write_entry():
    entry = input("Write your diary:\n")
    
    file = open(FILENAME, "a")  # append mode
    file.write(entry + "\n")
    file.close()
    
    print("Diary added!\n")
